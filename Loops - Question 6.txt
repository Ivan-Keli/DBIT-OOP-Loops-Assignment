public class Question6 {

    public static void main(String[] args) {
        // Starting number
        int start = 12;
        // Ending number
        int end = 28;
        // Step size
        int step = 2;

        // Loop to generate and output the numbers
        for (int i = start; i <= end; i += step) {
            System.out.print(i + " ");
        }
    }
}
