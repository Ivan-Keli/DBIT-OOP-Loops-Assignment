public class Question7 {

    public static void main(String[] args) {
        int start = 20;
        int end = 25;
        int sum = 0;

        // Loop to compute the sum
        for (int i = start; i <= end; i++) {
            sum += i;
        }

        System.out.println("The sum of numbers between " + start + " and " + end + " is: " + sum);
    }
}
